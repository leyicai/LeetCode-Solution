# 273. Integer to English Words
@(LeetCode)[String, Hard]

- Store all words in arrays
-  Use helper function `smallNumberToWords` to convert numbers <1000

```java
private final String[] LESS_THAN_20 = { "", "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine",
        "Ten", "Eleven", "Twelve", "Thirteen", "Fourteen", "Fifteen", "Sixteen", "Seventeen", "Eighteen",
        "Nineteen" };
private final String[] TENS = { "", "Ten", "Twenty", "Thirty", "Forty", "Fifty", "Sixty", "Seventy", "Eighty",
        "Ninety" };
private final String[] THOUSANDS = { "", "Thousand", "Million", "Billion" };

public String numberToWords(int num) {
    if (num == 0)
        return "Zero";
    String res = "";
    int i = 0;
    while (num > 0) {
        if (num % 1000 != 0) {
            res = smallNumberToWords(num % 1000) + THOUSANDS[i] + " " + res;
        }
        i++;
        num /= 1000;
    }
    return res.trim();
}

private String smallNumberToWords(int num) {
    if (num == 0)
        return "";
    if (num < 20)
        return LESS_THAN_20[num] + " ";
    if (num < 100)
        return TENS[num / 10] + " " + smallNumberToWords(num % 10);
    return LESS_THAN_20[num / 100] + " Hundred " + smallNumberToWords(num % 100);
}
```
