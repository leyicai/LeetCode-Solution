26&80. Remove Duplicates from Sorted Array

When duplicates are less than k, pointer i and j moves together. Duplicates more than k times (pointed by j) will be removed and pointer is tracking the length of array

1.int removeDuplicates(vector<int>& nums,int k) {

2.    if(nums.size()<k) return nums.size(); // if array size is less than k then return the same

3.    int i,j;

4.    for(i=k,j=k ; j<nums.size();j++)

5.        if(nums[i-k]!=nums[j]) 

6.            nums[i++]=nums[j];

7.    return i;

8.}
