Compare leftSum, rightSum and crossSum
For crossSum: add numbers from middle and record the max sum of two half

5.private int maxSubArrayHelper(int[] nums, int left, int right) {

6.    if (left == right)

7.        return nums[left];

8.    int mid = (left + right) / 2;

9.    int leftSum = maxSubArrayHelper(nums, left, mid);

10.    int rightSum = maxSubArrayHelper(nums, mid + 1, right);

11.    int crossSum = crossSubArray(nums, left, right);

12.    return Math.max(leftSum, Math.max(rightSum, crossSum));

13.}

15.private int crossSubArray(int[] nums, int left, int right) {

16.    int leftSum = Integer.MIN_VALUE, rightSum = Integer.MIN_VALUE, sum = 0;

17.    int mid = (left + right) / 2;

18.    for (int i = mid; i >= left; i--) {

19.        sum += nums[i];

20.        leftSum = Math.max(leftSum, sum);

21.    }

22.    sum = 0;

23.    for (int i = mid + 1; i <= right; i++) {

24.        sum += nums[i];

25.        rightSum = Math.max(rightSum, sum);

26.    }

27.    return leftSum + rightSum;

28.}
