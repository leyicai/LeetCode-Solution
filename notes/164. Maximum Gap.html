164. Maximum Gap

bucket sort

  • bucket size: (max - min)/length + 1

    total gap = max - min
    n numbers have n-1 gap: average gap = total gap / (length-1)
    maxGap >= average gap
    bucket size =

  • bucket number: length / bucket size + 1 = length - 1

  • keep two arrays of max and min in each bucket. Since there are len-2 numbers that are not equal min or max and there are len-1 buckets, at least one of the buckets are empty.

    • maxGap must be one of the diff between min of previous and max of next non-empty buckets.

  • O(n) space and time

1.class Solution {

2.    public int maximumGap(int[] nums) {

3.        if (nums == null || nums.length < 2)

4.            return 0;

5.        // Get max and min of the array

6.        int min = Integer.MAX_VALUE, max = Integer.MIN_VALUE;

7.        for (int n : nums) {

8.            min = Math.min(min, n);

9.            max = Math.max(max, n);

10.        }

11.        // Assign buckets

12.        // We only care about the min and max in each bucket

13.        int gap = (max - min) / nums.length + 1, bucketNum = (max - min) / gap + 1;

14.        int[] bucketMin = new int[bucketNum];

15.        int[] bucketMax = new int[bucketNum];

16.        Arrays.fill(bucketMin, Integer.MAX_VALUE);

17.        Arrays.fill(bucketMax, Integer.MIN_VALUE);

18.        // Fill the buckets with numbers in array

19.        for (int n : nums) {

20.            int i = (n - min) / gap;

21.            bucketMin[i] = Math.min(bucketMin[i], n);

22.            bucketMax[i] = Math.max(bucketMax[i], n);

23.        }

24.        // Scan the buckets to get the maximum gap

25.        gap = 0;

26.        for (int i = 0; i < bucketNum; i++) {

27.            if (bucketMin[i] != Integer.MAX_VALUE) {

28.                gap = Math.max(gap, bucketMin[i] - min);

29.                min = bucketMax[i];

30.            }

31.        }

32.        return gap;

33.    }

34.}
