6.public void rotate(int[] nums, int k) {

7.    k %= nums.length;

8.    reverse(nums, 0, nums.length - 1);

9.    reverse(nums, 0, k - 1);

10.    reverse(nums, k, nums.length - 1);

11.}

13.public void reverse(int[] nums, int start, int end) {

14.    while (start < end) {

15.        int temp = nums[start];

16.        nums[start] = nums[end];

17.        nums[end] = temp;

18.        start++;

19.        end--;

20.    }
